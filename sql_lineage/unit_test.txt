SELECT
    oi.order_id,
    oi.product_id,
    oi.quantity,
    oi.unit_price,
    oi.discount_pct,

    -- line revenue (before discount)
    oi.quantity * oi.unit_price                                        AS gross_line_amount,

    -- discount applied
    oi.quantity * oi.unit_price * (oi.discount_pct / 100.0)          AS discount_amount,

    -- net revenue after discount
    oi.quantity * oi.unit_price * (1 - oi.discount_pct / 100.0)     AS net_line_amount,

    -- gross margin %  (cost from products table)
    (oi.unit_price - p.unit_cost) / NULLIF(oi.unit_price, 0) * 100  AS margin_pct,

    -- tax (8.5%)
    oi.quantity * oi.unit_price * 0.085                              AS tax_amount
FROM  order_items oi
JOIN  products    p  ON oi.product_id = p.product_id;
